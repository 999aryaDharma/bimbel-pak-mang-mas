function handlePreview(input, tahapId) {
  const file = input.files[0];
  const previewContainer = document.getElementById(`preview_${tahapId}`);
  const btnAI = document.getElementById(`btn_ai_${tahapId}`);

  if (file) {
    const reader = new FileReader();
    reader.onload = function(e) {
      if (previewContainer) {
        previewContainer.innerHTML = `<img src="${e.target.result}" style="max-width: 100%; border-radius: 10px; border: 2px solid #6366f1; margin-top: 10px;">`;
      }
      // Reset tombol menjadi biru kembali
      if (btnAI) {
        btnAI.disabled = false;
        btnAI.innerHTML = `✨ Analisa AI ${tahapId.toUpperCase()}`;
        btnAI.style.background = "#6366f1";
      }
    };
    reader.readAsDataURL(file);
  }
}

let HTP_DATA_CACHE = {};

const HTP_COLORS = { 
  "Rendah": "#ef4444", 
  "Cukup": "#f59e0b", 
  "Baik": "#10b981", 
  "Sangat Baik": "#6366f1" 
};

// ==========================================
// 2. FUNGSI ANALISA UTAMA (AI)
// ==========================================
async function jalankanAI_HTP(tahapId) {
  const btn = document.getElementById(`btn_ai_${tahapId}`);
  const previewImg = document.querySelector(`#preview_${tahapId} img`);
  
  if (!previewImg) {
    alert("Silakan upload gambar terlebih dahulu!");
    return;
  }

  if (btn) {
    btn.innerHTML = "⌛ Memproses...";
    btn.disabled = true;
    btn.style.background = "#94a3b8"; 
  }

  try {
    google.script.run
      .withSuccessHandler(function(response) {
        if (response.error) {
          alert("Error: " + response.error);
          if (btn) btn.innerHTML = "❌ Gagal";
          btn.disabled = false;
          return;
        }

        // Simpan database referensi dari server ke cache client
        HTP_DATA_CACHE[tahapId] = {
          narasi: response.db_referensi ? response.db_referensi.narasi : null,
          kesimpulan: response.db_referensi ? response.db_referensi.kesimpulan : null
        };

        // Update Skor & Narasi Per Aspek
        const aspekIds = ['pro', 'detl', 'pers', 'line'];
        aspekIds.forEach(aspek => {
          const elS = document.getElementById(`skor_${tahapId}_${aspek}`);
          const elN = document.getElementById(`narasi_detail_${tahapId}_${aspek}`);
          
          if (elS) elS.value = response.skor[aspek] || 0;
          if (elN) elN.value = response.narasi[aspek] || "";
        });

        // Jalankan perhitungan total untuk update header & kesimpulan
        hitungTotalPerTahap(tahapId);

        if (btn) {
          btn.innerHTML = "✅ Sudah Dianalisa";
          btn.style.background = "#10b981";
          btn.disabled = false;
        }
      })
      .withFailureHandler(function(err) {
        console.error("Gagal server:", err);
        if (btn) {
          btn.innerHTML = "❌ Error Server";
          btn.disabled = false;
        }
      })
      .panggilAI_HTP(previewImg.src, tahapId); 

  } catch (error) {
    console.error("Client Error:", error);
    if (btn) btn.innerHTML = "✨ Analisa AI";
  }
}

  aspekIds.forEach(aspek => {
    const elSkor = document.getElementById(`skor_${tahapId}_${aspek}`);
    const elNarasiBox = document.getElementById(`narasi_detail_${tahapId}_${aspek}`);
    
    if (elSkor) {
      let nilai = parseInt(elSkor.value) || 0;
      if (nilai > 5) { nilai = 5; elSkor.value = 5; } 
      total += nilai;
      
      // Update Narasi Detail (Manual Edit Support)
      if (elNarasiBox) {
        let teks = "";
        if (HTP_DATA_CACHE[tahapId] && HTP_DATA_CACHE[tahapId].narasi) {
          teks = HTP_DATA_CACHE[tahapId].narasi[aspek][nilai] || "";
        } else {
          teks = getNarasiDariDB(tahapId, aspek, nilai);
        }
        if (teks) elNarasiBox.value = teks;
      }
    }
  });

  const kategori = total >= 17 ? "Sangat Baik" : total >= 13 ? "Baik" : total >= 9 ? "Cukup" : "Rendah";
  
  // Update Tampilan Atas (Header)
  updateTampilanHeader(tahapId, total, kategori);

  // Update 4 Kotak Kesimpulan (Teknis, Konten, Simbolis, Observasi)
  let dataKes;
  if (HTP_DATA_CACHE[tahapId] && HTP_DATA_CACHE[tahapId].kesimpulan) {
    dataKes = HTP_DATA_CACHE[tahapId].kesimpulan[kategori];
  } else {
    dataKes = getKesimpulanDariDB(tahapId, kategori);
  }

  const fields = ['teknis', 'konten', 'simbolis', 'observasi'];
  fields.forEach(f => {
    const el = document.getElementById(`${f}_${tahapId}`);
    if (el) el.value = (dataKes && dataKes[f]) ? dataKes[f] : "-";
  });

  // Update Rekapitulasi Paling Bawah
  hitungTotalKeseluruhan();
}

function updateTampilanHeader(tahapId, total, kategori) {
  const elTotal = document.getElementById(`total_score_${tahapId}`);
  const elKategori = document.getElementById(`kategori_${tahapId}`);

  if (elTotal) elTotal.value = total;
  if (elKategori) {
    elKategori.value = kategori;
    elKategori.style.color = HTP_COLORS[kategori] || "#1e293b";
  }
}

function getNarasiDariDB(tahapId, aspekId, nilai) {
  const mapping = { 'orang': dbNarasiOrang, 'rumah': dbNarasiRumah, 'pohon': dbNarasiPohon, 'gabung': dbNarasiGabungan };
  const db = mapping[tahapId.toLowerCase()];

}

function getKesimpulanDariDB(tahapId, kategori) {
  const mapping = { 'orang': dbKesimpulanOrang, 'rumah': dbKesimpulanRumah, 'pohon': dbKesimpulanPohon, 'gabung': dbKesimpulanGabungan };
  const db = mapping[tahapId.toLowerCase()];
  return (db && db[kategori]) ? db[kategori] : { teknis: "-", konten: "-", simbolis: "-", observasi: "-" };
}

function hapusPeserta(id) {
  if (confirm("⚠️ PERINGATAN!\n\nApakah Anda yakin ingin menghapus permanen data peserta dengan ID: " + id + "?\nData yang dihapus tidak dapat dikembalikan.")) {
    
    // Tampilkan status loading sederhana
    const btn = event.target;
    const originalText = btn.innerText;
    btn.disabled = true;
    btn.innerText = "⏳...";

    google.script.run.withSuccessHandler(res => {
      if (res.success) {
        alert("✅ " + res.message);
        renderPeserta(); // Segarkan tabel agar data yang dihapus hilang
        renderHome();    // Update jumlah peserta di dashboard
      } else {
        alert("❌ Gagal: " + res.message);
        btn.disabled = false;
        btn.innerText = originalText;
      }
    }).hapusDataPesertaSheet(id);
  }
}
