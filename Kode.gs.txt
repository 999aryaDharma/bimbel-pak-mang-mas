function generateIdTest(prefix) {
  const tgl = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyyMMdd");
  const unique = Math.floor(Math.random() * 1000);
  return `${prefix}-${tgl}-${unique}`;
}

function getPesertaDropdown() {
  const sheet = getSS().getSheetByName(CONFIG.SHEET.PESERTA);
  const data = sheet.getDataRange().getValues();
  
  // Ambil data mulai dari baris kedua (slice 1)
  // Pastikan kolom 1 adalah ID dan kolom 2 adalah NAMA
  return data.slice(1).map(r => ({ 
    id: r[1],   // Sesuaikan indeks kolom (0, 1, 2...)
    nama: r[2] 
  })); 
}

function uploadFile(base64, filename) {
  if (!base64) return "-";
  const blob = Utilities.newBlob(Utilities.base64Decode(base64), "image/jpeg", filename);
  const file = DriveApp.getFolderById(CONFIG.FOLDER_ID).createFile(blob);
  return "https://drive.google.com/uc?export=view&id=" + file.getId();
}

// Fungsi simulasi AI untuk menilai Gambar Wartegg
function analisaWarteggAI() {
  // Dalam pengembangan nyata, di sini Anda bisa menghubungkan ke API Gemini atau OpenAI
  // Untuk saat ini, kita buat AI memberikan nilai default yang optimal (Random High Score)
  
  const nilaiAI = [];
  for(let i=0; i<8; i++) {
    nilaiAI.push(Math.floor(Math.random() * (5 - 3 + 1)) + 3); // Memberikan nilai acak 3-5
  }
  
  const total = nilaiAI.reduce((a, b) => a + b, 0);
  
  return {
    nilai: nilaiAI,
    total: total
  };
}

    const copy = templateDoc.makeCopy("HTP_" + data.Nama, folderLaporan);
    const doc = DocumentApp.openById(copy.getId());
    const body = doc.getBody();

    body.replaceText("{{Nama}}", data.Nama);

    return { success: true, pdfUrl: pdf.getUrl() };

  } catch (err) {
    return { success: false, message: err.toString() };
  }
}

function getSemuaPeserta() {
  try {
    const ss = getSS(); // Menggunakan fungsi helper yang sudah Anda buat di atas
    const sheet = ss.getSheetByName(CONFIG.SHEET.PESERTA);
    
    if (!sheet) {
      return { success: false, message: "Sheet DATA_PESERTA tidak ditemukan" };
    }

    const data = sheet.getDataRange().getValues();
    
    if (data.length <= 1) {
      return { success: true, rows: [], header: [], total: 0 };
    }

    // Ambil header (Abaikan kolom A/Timestamp, ambil B sampai akhir)
    const header = data[0].slice(1); 
    
    // Ambil data (Abaikan kolom A/Timestamp)
    const rows = data.slice(1).map((row, index) => {
      let cleanedRow = row.slice(1).map(cell => {
        if (cell instanceof Date) {
          return Utilities.formatDate(cell, "GMT+7", "dd/MM/yyyy");
        }
        return cell === "" ? "-" : cell;
      });
      // Tambahkan index baris asli (untuk keperluan hapus data nanti)
      return {
        data: cleanedRow,
        id: row[1] // ID_PESERTA biasanya ada di kolom B (index 1)
      };
    });
    
    return {
      success: true,
      header: header,
      rows: rows,
      total: rows.length
    };
  } catch (e) {
    return { success: false, message: e.toString() };
  }
}

function hapusDataPesertaSheet(id) {
  try {
    const ss = getSS();
    const sheet = ss.getSheetByName(CONFIG.SHEET.PESERTA);
    const data = sheet.getDataRange().getValues();
    
    // Cari di kolom kedua (indeks 1) karena ID Peserta biasanya setelah Timestamp
    for (let i = 1; i < data.length; i++) {
      if (data[i][1] == id) { 
        sheet.deleteRow(i + 1); 
        return { success: true, message: "Data peserta " + id + " berhasil dihapus." };
      }
    }
    return { success: false, message: "ID tidak ditemukan di database." };
  } catch (e) {
    return { success: false, message: e.toString() };
  }
}

/**
 * Fungsi Utama untuk Analisa HTP (House, Tree, Person)
 * @param {string} imageData - Data base64 gambar (opsional untuk simulasi)
 * @param {string} tahapId - ID tahap (orang, rumah, pohon, atau gabung)
 */
/**
 * Fungsi Utama untuk Analisa HTP (House, Tree, Person)
 * Diperbarui agar mengirimkan database narasi ke Client
 */
function panggilAI_HTP(imageData, tahapId) {
  try {
    const idSesuai = tahapId.toLowerCase();
    
    // Mapping Database berdasarkan tahapId
    const dbMapping = {
      'orang': { narasi: dbNarasiOrang, kesimpulan: dbKesimpulanOrang },
      'rumah': { narasi: dbNarasiRumah, kesimpulan: dbKesimpulanRumah },
      'pohon': { narasi: dbNarasiPohon, kesimpulan: dbKesimpulanPohon },
      'gabung': { narasi: dbNarasiGabungan, kesimpulan: dbKesimpulanGabungan }
    };

    const aktif = dbMapping[idSesuai];
    if (!aktif) throw new Error("Database untuk " + tahapId + " tidak ditemukan.");

    const aspekIds = ['pro', 'detl', 'pers', 'line'];
    let skorHasil = {};
    let totalScore = 0; 
    let narasiHasil = {};

    // Proses Perhitungan Skor & Narasi (Simulasi AI)
    aspekIds.forEach(id => {
      const nilai = Math.floor(Math.random() * 5) + 1;
      skorHasil[id] = nilai;
      totalScore += nilai; 
      
      if (aktif.narasi[id] && aktif.narasi[id][String(nilai)]) {
        narasiHasil[id] = aktif.narasi[id][String(nilai)];
      } else {
        narasiHasil[id] = `Analisa aspek ${id} dengan skor ${nilai}.`;
      }
    });

    // Penentuan Kategori
    let kategori = "";
    if (totalScore <= 8) kategori = "Rendah";
    else if (totalScore <= 12) kategori = "Cukup";
    else if (totalScore <= 16) kategori = "Baik";
    else kategori = "Sangat Baik";

    const kesimpulanHasil = aktif.kesimpulan[kategori] || {
      teknis: "-", konten: "-", simbolis: "-", observasi: "-"
    };

    // --- BAGIAN PENTING UNTUK OTOMATISASI ---
    // Return data lengkap beserta referensi database agar Client bisa update real-time
    return {
      tahapId: tahapId,
      skor: skorHasil,
      total: totalScore,
      kategori: kategori,
      narasi: narasiHasil,
      kesimpulan: kesimpulanHasil,
      // Tambahkan ini: Mengirim data referensi ke Browser
      db_referensi: {
        narasi: aktif.narasi,
        kesimpulan: aktif.kesimpulan
      }
    };

  } catch (err) {
    console.error("Error panggilAI_HTP:", err.message);
    return { error: err.message };
  }
}
